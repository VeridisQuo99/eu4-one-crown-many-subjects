	icon = "paper_with_seal"
	lock_level_when_selected = yes
	allow_normal_conversion = no
	legacy_equivalent = noble_republic
	potential = {
		OR = {
			has_reform = royal_protectorate_government
			have_had_reform = royal_protectorate_government
		}
	}
	modifiers = {
		# Unrest Boost
		tolerance_heathen = 1
		tolerance_of_heathens_capacity = 1
		global_rebel_suppression_efficiency = 0.10
		years_of_nationalism = -5
		global_unrest = -2
		no_religion_penalty = yes	

		# Economy Boost
		global_tax_modifier = 0.025
		production_efficiency = 0.025
		trade_efficiency = 0.025
		
		# Government Boost
		governing_capacity_modifier = 0.025

		# Military & Navy Boost
		fire_damage_received = -0.025
		shock_damage_received = -0.025
		morale_damage_received = -0.025
		global_manpower_modifier = 0.025
		global_sailors_modifier = 0.025
		land_forcelimit_modifier = 0.025
		naval_forcelimit_modifier = 0.025
		# has_voc_indiamen = yes
		# allowed_voc_indiamen_fraction = 1
	}
	nation_designer_cost = 200
	valid_for_nation_designer = yes
	boost_income = yes
	duration = 4
	
	factions = {
		mr_aristocrats
		mr_traders
		mr_guilds
	}

	custom_attributes = {
		locked_government_type = yes
		enables_plutocratic_idea_group = yes
		enables_aristocratic_idea_group = no
		enables_estate_burghers = yes
		enables_estate_nobles = yes
		enables_estate_clergy = yes
		is_merchant_republic = yes
		elections_increase_factions_influence = yes
	}

	conditional = {
		allow = {
			has_dlc = "Leviathan"
		}
		allow_draft_transport_ships = yes
	}
	conditional = {
		allow = {
			has_dlc = "Res Publica"
			NOT = { has_dlc = "Mare Nostrum" }
		}
		can_use_trade_post = yes
	}
	conditional = {
		allow = {
			has_dlc = "Mare Nostrum"
			NOT = { has_dlc = "Res Publica" }
		}
		can_form_trade_league = yes
		trade_city_reform = trading_city
	}
	conditional = {
		allow = {
			has_dlc = "Res Publica"
			has_dlc = "Mare Nostrum"
		}
		can_use_trade_post = yes
		can_form_trade_league = yes
		trade_city_reform = trading_city
	}

	conditional = {
		allow = {
			OR = {
				has_dlc = "Common Sense"
				has_dlc = "Domination"
			}
		}
		has_parliament = yes
		custom_attributes = {
			gain_rt_from_parliament = yes
		}
	}
}
