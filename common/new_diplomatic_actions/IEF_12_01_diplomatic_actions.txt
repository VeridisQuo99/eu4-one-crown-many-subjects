establish_crown_protector = {
	category = influence

	require_acceptance = no

	is_visible = { # ONLY FOR PLAYERS
		ROOT = {
			ai = no
		}
	}

	is_allowed = { 
		FROM = { 
			is_subject_of = ROOT # NEED TO BE OUR SUBJECT
			NOT = {
				OR = {
					#vanilla
					# is_subject_of_type = vassal
					# is_subject_of_type = march
					is_subject_of_type = daimyo_vassal
					# is_subject_of_type = appanage
					# is_subject_of_type = personal_union
					# is_subject_of_type = client_vassal
					# is_subject_of_type = client_march
					is_subject_of_type = colony
					is_subject_of_type = crown_colony
					is_subject_of_type = private_enterprise
					is_subject_of_type = self_governing_colony
					is_subject_of_type = tributary_state
					# is_subject_of_type = eyalet
					# is_subject_of_type = core_eyalet
					is_subject_of_type = commercial_enterprise
					is_subject_of_type = trade_protectorate
					# is_subject_of_type = hereditary_pronoia_subject_type
					# is_subject_of_type = pronoia_subject_type
					is_subject_of_type = cultural_sphere_of_influence_subject
					is_subject_of_type = theodoro_personal_union
					# is_subject_of_type = incorporated_vassal
					is_subject_of_type = nahuatl_tributary
					is_subject_of_type = european_colony
					# is_subject_of_type = mongol_brother_realm
					# is_subject_of_type = integrated_personal_union
					# is_subject_of_type = soyurghal_subject_type

					#additives
					is_subject_of_type = crown_protector
				}
			}
			NOT = { # MUST BE LOYAL TO THE CROWN
				liberty_desire = 50 
			}
			has_opinion = { # OFC MUST HAVE POSITIVE OPINION
				who = ROOT
				value = 0
			}
			NOT = { # MUST BE SMALLER THAN US
				development_of_overlord_fraction = 100.00 
			}
		}
		ROOT = {
			adm_power = 5
			dip_power = 5
			mil_power = 5
	
			government_rank = 2
		}
	}

	on_accept = {
		if = {
			limit = {
				FROM = { 
					is_subject_of = ROOT # NEED TO BE OUR SUBJECT
					NOT = { # MUST BE LOYAL TO THE CROWN
						liberty_desire = 50 
					}
					has_opinion = { # OFC MUST HAVE POSITIVE OPINION
						who = ROOT
						value = 0
					}
					NOT = { # MUST BE SMALLER THAN US
						development_of_overlord_fraction = 100.00 
					}
				}
				NOT = { # CAN'T repeat
					FROM = { is_subject_of_type = crown_protector }
				}
				ROOT = {
					adm_power = 5
					dip_power = 5
					mil_power = 5
				}
			}

			FROM = {
				grant_independence = yes
			}

			create_subject = {
				subject = FROM
				subject_type = crown_protector
			}

			ROOT ={
				add_adm_power = -5
				add_dip_power = -5
				add_mil_power = -5
			}

			FROM = {
				add_adm_power = 10
				add_dip_power = 10
				add_mil_power = 10
				add_treasury = 10
				set_prestige_to_zero = yes
			}	
		}
	}

	ai_will_do = { # this is a trigger
		always = no
	}
}

# Isolate Crown Protector (Hem Savaş Hem İttifak Haklarını Alır)
isolate_crown_protector = {
	category = influence
	require_acceptance = no

	is_visible = {
		FROM = {
			is_subject_of = ROOT
			is_subject_of_type = crown_protector
		}
	}
	is_allowed = {
		always = yes
	}
	on_accept = {
		FROM = {
			add_country_modifier = {
				name = restricted_diplomatic_rights_royal
				duration = -1
			}
			add_opinion = {
				who = ROOT
				modifier = isolated_by_overlord
			}
			every_ally = { break_alliance = PREV }
		}

		create_subject = {
			subject = FROM
			subject_type = peaceful_crown_protector
		}
	}
	ai_will_do = { always = no }
}

# Empower Crown Protector (Tüm Hakları Geri Verir)
empower_crown_protector = {
	category = influence
	require_acceptance = no

	is_visible = {
		FROM = {
			is_subject_of = ROOT
			is_subject_of_type = peaceful_crown_protector
		}
	}
	is_allowed = {
		always = yes
	}
	on_accept = {
		FROM = {
			remove_country_modifier = restricted_diplomatic_rights_royal
		}

		create_subject = {
			subject = FROM
			subject_type = crown_protector
		}
	}
	ai_will_do = { always = no }
}
